Gerador de Senha

import java.util.Random;
import java.util.Scanner;

public class GeradorSenha {
    public static final String CARACTERES = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%&*()_+-={}|[]:;<>?,./";

    public static void main(String[] args) {
        System.out.println("Boa tarde, aqui vc escolhera quantos caracteres gostaria em sua senha randomizada que o banco te oferece:");
        Scanner scanner = new Scanner(System.in);
        int quantidade = scanner.nextInt();
        String senha = gerarSenha(quantidade);
        System.out.println("Senha gerada: " + senha);
        scanner.close();
    }

    public static String gerarSenha(int quantidade) {
        Random random = new Random();
        StringBuilder senha = new StringBuilder();

        for (int i = 0; i < quantidade; i++) {
            char caractere = CARACTERES.charAt(random.nextInt(CARACTERES.length()));
            senha.append(caractere);
        }

        return senha.toString();
    }
}
